class Checker{
    Comparator<Player> desc;
    
    public Checker(){
        desc = new Comparator<Player>(){
            public int compare(Player p1, Player p2){
                if(p1.score != p2.score){
                    return p2.score - p1.score;
                }
                else{
                    return p2.name.compareTo(p1.name);
                }
            }
        };
    }
}
